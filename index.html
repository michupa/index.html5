break; } } if (!isLocationValid) { return sendResponse("error", "指定されたロケーションがマスターに登録されていません。"); } // 重複チェック（通常の型式スキャン時のみ。手入力の一時ID時はスルー） if (!tempId) { const dataValues = dataSheet.getDataRange().getValues(); for (let i = 1; i < dataValues.length; i++) { if (dataValues[i][1] === modelCode) { // B列:型式コード return sendResponse("error", "この型式は既に登録されています。"); } } } // Dataシートへの書き込み // A列:入庫日, B列:型式コード, C列:一時ID, D列:ロケーション, E列:空(予備), F列:製品名・メモ dataSheet.appendRow([timestamp, modelCode, tempId, location, "", productName]); let foundIndex = -1; for (let i = 1; i < dataValues.length; i++) { const isModelMatch = (!tempId && dataValues[i][1] === modelCode); const isTempIdMatch = (tempId && dataValues[i][2] === tempId); if ((isModelMatch || isTempIdMatch) && dataValues[i][3] === location) { foundIndex = i; break; } } if (foundIndex === -1) { return sendResponse("error", "該当する製品が指定のロケーションに見つかりません。"); } // 出荷シートへ記録してDataシートから削除 shippingSheet.appendRow([timestamp, modelCode, tempId, location, "", dataValues[foundIndex][5] || productName]); const dataValues = dataSheet.getDataRange().getValues(); let foundIndex = -1; for (let i = 1; i < dataValues.length; i++) { const isModelMatch = (!tempId && dataValues[i][1] === modelCode); const isTempIdMatch = (tempId && dataValues[i][2] === tempId); if ((isModelMatch || isTempIdMatch) && dataValues[i][3] === location) { foundIndex = i; break; } } if (foundIndex === -1) { return sendResponse("error", "移動対象の製品が指定の移動元ロケーションに見つかりません。"); } // D列（ロケーション）を更新 dataSheet.getRange(foundIndex + 1, 4).setValue(newLocation); // 履歴シートへの書き込み historySheet.appendRow([timestamp, location, modelCode || tempId, newLocation, dataValues[foundIndex][5] || productName]); return sendResponse("success", "配置替え処理が完了しました。"); } // ④ ロケーション登録 else if (mode.includes("ロケーション登録")) { if (!location) return sendResponse("error", "ロケーション名が空です。");
